'เพิ่มผู้ใช้ใหม่' : 'แก้ไขผู้ใช้'; document.getElementById('userId').value = id || ''; if (mode === 'add') { document.getElementById('userForm').reset(); document.getElementById('passwordGroup').style.display = 'block'; // Always show for add document.getElementById('passwordHelp').style.display = 'none'; } else { document.getElementById('passwordGroup').style.display = 'block'; document.getElementById('passwordHelp').style.display = 'block'; // Optional for edit // Pre-fill logic handled in editUser } modal.style.display = 'block'; } function closeModal() { document.getElementById('userModal').style.display = 'none'; } window.editUser = function(id, username, email, role) { openModal('edit', id); document.getElementById('username').value = username; document.getElementById('email').value = email; document.getElementById('role').value = role; // Username is typically immutable or primary, but here we allow editing fields except ID // Ideally username shouldn't change if it's a key, but depends on DB. We'll disable username edit for safety document.getElementById('username').disabled = true; } window.deleteUser = async function(id) { if(!confirm('ยืนยันลบผู้ใช้งาน?')) return; try { const res = await fetch(`/api/users/${id}`, { method: 'DELETE', headers: { 'Authorization': `Bearer ${token}` } }); if(res.ok) fetchUsers(); else { const d = await res.json(); alert(d.message); } } catch(e) { console.error(e); } } document.querySelector('.btn-login').addEventListener('click', () => { openModal('add'); document.getElementById('username').disabled = false; }); document.getElementById('userForm').addEventListener('submit', async (e) => { e.preventDefault(); const id = document.getElementById('userId').value; const username = document.getElementById('username').value; const email = document.getElementById('email').value; const role = document.getElementById('role').value; const password = document.getElementById('password').value; const body = { username, email, role }; if (password) body.password = password; try { let res; if (id) { res = await fetch(`/api/users/${id}`, { method: 'PUT', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${token}` }, body: JSON.stringify(body) }); } else { res = await fetch('/api/users', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${token}` }, body: JSON.stringify(body) }); } if (res.ok) { closeModal(); fetchUsers(); } else { alert('Error saving user'); } } catch (e) { console.error(e); } }); fetchUsers();
